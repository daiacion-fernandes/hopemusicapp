function Aniversariantes({data,setData}) {

  const del=id=>setData(d=>({...d,aniversariantes:d.aniversariantes.filter(x=>x.id!==id)}));

  const now=new Date(); const thisMonth=now.getMonth()+1;

  const sorted=[...data.aniversariantes].sort((a,b)=>{const[,am,,ad]=a.data.split("-").map(Number);const[,bm,,bd]=b.data.split("-").map(Number);return am!==bm?am-bm:ad-bd;});

  const mesAtual=sorted.filter(a=>parseInt(a.data.split("-")[1])===thisMonth);
